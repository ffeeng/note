
- 遍历方式
```js

const copy = function (origin){
    if(typeof origin !=='object') return origin;
    let res = origin instanceof Array?[]:{};
    for(let key in origin){
        if(origin.hasOwnPropertyKey(key)){
            res[key] = origin[key]
        }   
    }   
    return res;
}

```

### 深拷贝

```js
}
```
